with first_log as (
    select
        player_id,
        min(event_date) as 'first_date'
    from
        Activity
    group by
        player_id
),
returns as (
    select
        f.player_id
    from
        first_log f
    where exists (
        select
            1
        from
            Activity a2
        where
            a2.player_id = f.player_id
        and
            a2.event_date = adddate(f.first_date, interval 1 day)
    )
)

select round(count(*) / (select count(*) from first_log),2) as fraction
from returns
